	\begin{lstlisting}
    fun ArrayR (a:array [0,n) of real, i:nat, j:nat)
        var s,m1,m2,k : nat
        {0 <= i <= j < n}
        s:= i + j
        if 
            s mod 2 = 0 -> m1,m2:= s/2 - 1, s/2+1
            [] s mod 2 = 1 -> m1,m2:= (s - 1)/2, (s+1)/2
        fi
        if
            i = j ->  skip
        []

        fi

    fun SUM (a:array [0,n) of real, i:nat, j:nat, acc:nat)

            i = j -> r := acc;  skip
